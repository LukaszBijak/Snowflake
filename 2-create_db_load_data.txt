CREATE DATABASE exercise_db;

CREATE TABLE CUSTOMERS 
(
   ID INTEGER NOT NULL, 
   FIRST_NAME VARCHAR, 
   LAST_NAME VARCHAR, 
   EMAIL VARCHAR, 
   AGE INTEGER, 
   CITY VARCHAR);

COPY INTO CUSTOMERS
FROM s3://snowflake-assignments-mc/gettingstarted/customers.csv 
FILE_FORMAT = (TYPE = csv, skip_header=1);

SELECT * FROM CUSTOMERS
